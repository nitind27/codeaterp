Authentication APIs

POST /api/auth/login Login
curl -X POST http://localhost:3000/api/auth/login \ -H "Content-Type: application/json" \ -d '{"email": "user@example.com", "password": "password123"}'
GET /api/auth/me Get Current User
curl -X GET http://localhost:3000/api/auth/me \ -H "Authorization: Bearer YOUR_TOKEN_HERE"
POST /api/auth/register Register User (Admin/HR)
curl -X POST http://localhost:3000/api/auth/register \ -H "Content-Type: application/json" \ -H "Authorization: Bearer YOUR_TOKEN_HERE" \ -d '{"email": "newuser@example.com", "password": "password123", "role": "employee", "employeeData": {"first_name": "Jane", "last_name": "Smith", "phone": "1234567890", "department": "HR", "designation": "HR Manager"}}'
